# Oceanheart UI Cleanup Commands

## IMPORTANT: Execute these commands manually in your terminal after review

# 1. Create backup branch (already done)
git checkout -b cleanup-code

# 2. Remove Aider-related files
git rm -f aider.session bak.aider.model.settings.yml session.aider reasoning-effort

# 3. Remove duplicate Next.js files from root directory
git rm -f error.tsx not-found.tsx page.tsx globals.css opengraph-image.png twitter-image.png

# 4. Remove temporary and test files
git rm -f names.db blog.prompt.xml cxo.html

# 5. Remove orphaned directories
# Note: Check if these directories exist before removing
rm -rf signin/
rm -rf privacy-policy/
rm -rf tos/
rm -rf src/

# 6. Consolidate tasks/ and tasks-refactor/ directories
# After reviewing both directories, remove the duplicate one:
# rm -rf tasks-refactor/  # Only if tasks/ is the primary one

# 7. Move documentation files to appropriate directories
# Create necessary directories first
mkdir -p docs/refactor
mkdir -p docs/development
mkdir -p docs/database

# Then move files
git mv CODE_REVIEW.md docs/
git mv DEAD_CODE_TARGETS.md docs/refactor/
git mv gitflow.mdc docs/development/
git mv erd.md docs/database/
git mv spec.md docs/

# 8. Finalize ARCHITECTURE.md
git mv NEW_ARCHITECTURE.md ARCHITECTURE.md  # If NEW_ARCHITECTURE.md is meant to replace the existing one

# 9. Commit changes with descriptive messages
git commit -m "chore: remove aider-related files"
git commit -m "chore: remove duplicate Next.js files from root"
git commit -m "chore: remove temporary and test files"
git commit -m "chore: remove orphaned directories"
git commit -m "chore: consolidate tasks directories"
git commit -m "docs: reorganize documentation files"
git commit -m "docs: update ARCHITECTURE.md"

# 10. Final verification and testing
bun install  # Ensure dependencies are up to date
bun run build  # Verify the build still works

# 11. Push the changes
git push origin cleanup-code

# Then create a pull request to merge these changes into the main branch
